    2. Create server/catalog/scanner.py:
       ```python
       import asyncio
       from pathlib import Path
       from concurrent.futures import ThreadPoolExecutor
       import uuid
       import numpy as np
       import nibabel as nib
       from catalog.models import VolumeMetadata
       from catalog.dicom_grouper import group_dicom_files

       KNOWN_NON_DICOM_EXTENSIONS = {'.txt', '.xml', '.json', '.py', '.csv', '.md', '.html', '.log', '.cfg', '.ini', '.yaml', '.yml', '.toml', '.png', '.jpg', '.jpeg', '.gif', '.bmp', '.pdf', '.zip', '.tar', '.gz'}
       NIFTI_EXTENSIONS = {'.nii', '.nii.gz'}

       def is_nifti_file(filepath: Path) -> bool:
           name = filepath.name.lower()
           return name.endswith('.nii') or name.endswith('.nii.gz')

       def is_dicom_file(filepath: Path) -> bool:
           ext = filepath.suffix.lower()
           if ext in ('.dcm',):
               return True
           if ext in KNOWN_NON_DICOM_EXTENSIONS:
               return False
           # Extensionless or unknown extension: check DICOM preamble magic bytes
           try:
               with open(filepath, 'rb') as f:
                   f.seek(128)
                   return f.read(4) == b'DICM'
           except (IOError, OSError):
               return False

       def guess_nifti_modality(data: np.ndarray) -> str:
           min_val = float(np.min(data))
           if min_val < -500:
               return "CT"
           return "MR"

       def _catalog_nifti(filepath: Path) -> VolumeMetadata:
           img = nib.load(str(filepath))
           header = img.header
           data = img.get_fdata(dtype=np.float32)
           dims = list(img.shape[:3])
           spacing = [float(s) for s in header.get_zooms()[:3]]
           dtype_str = str(header.get_data_dtype())
           modality = guess_nifti_modality(data)
           stat = filepath.stat()
           from datetime import datetime, timezone
           file_date = datetime.fromtimestamp(stat.st_mtime, tz=timezone.utc).strftime("%Y-%m-%d")
           return VolumeMetadata(
               id=str(uuid.uuid4()),
               filename=filepath.name,
               path=str(filepath),
               format="nifti",
               dimensions=dims,
               voxel_spacing=spacing,
               dtype=dtype_str,
               modality=modality,
               file_date=file_date,
           )

       async def scan_folder(folder: Path) -> list[VolumeMetadata]:
           loop = asyncio.get_event_loop()
           volumes = await loop.run_in_executor(None, _scan_folder_sync, folder)
           return volumes

       def _scan_folder_sync(folder: Path) -> list[VolumeMetadata]:
           nifti_files = []
           dicom_files = []
           for filepath in folder.rglob('*'):
               if not filepath.is_file():
                   continue
               if is_nifti_file(filepath):
                   nifti_files.append(filepath)
               elif is_dicom_file(filepath):
                   dicom_files.append(filepath)
           volumes: list[VolumeMetadata] = []
           for nf in nifti_files:
               try:
                   volumes.append(_catalog_nifti(nf))
               except Exception:
                   pass  # Skip unreadable files
           if dicom_files:
               dicom_volumes = group_dicom_files(dicom_files)
               volumes.extend(dicom_volumes)
           return volumes
       ```

    3. Create server/catalog/dicom_grouper.py:
       ```python
       from pathlib import Path
       import uuid
       import numpy as np
       import pydicom
       from catalog.models import VolumeMetadata

       def group_dicom_files(dicom_files: list[Path]) -> list[VolumeMetadata]:
           series_map: dict[str, list[Path]] = {}
           for f in dicom_files:
               try:
                   ds = pydicom.dcmread(str(f), stop_before_pixels=True)
                   series_uid = str(ds.SeriesInstanceUID)
                   series_map.setdefault(series_uid, []).append(f)
               except Exception:
                   continue
           volumes = []
           for series_uid, files in series_map.items():
               try:
                   vol = _build_dicom_volume_metadata(series_uid, files)
                   volumes.append(vol)
               except Exception:
                   continue
           return volumes

       def sort_dicom_slices(dicom_files: list[Path]) -> list[Path]:
           datasets = [(f, pydicom.dcmread(str(f), stop_before_pixels=True)) for f in dicom_files]
           ds0 = datasets[0][1]
           iop = [float(x) for x in ds0.ImageOrientationPatient]
           row_cosine = np.array(iop[:3])
           col_cosine = np.array(iop[3:])
           normal = np.cross(row_cosine, col_cosine)
           def slice_position(item):
               ds = item[1]
               ipp = np.array([float(x) for x in ds.ImagePositionPatient])
               return float(np.dot(ipp, normal))
           datasets.sort(key=slice_position)
           return [f for f, _ in datasets]

       def _build_dicom_volume_metadata(series_uid: str, files: list[Path]) -> VolumeMetadata:
           sorted_files = sort_dicom_slices(files) if len(files) > 1 else files
           ds = pydicom.dcmread(str(sorted_files[0]), stop_before_pixels=True)
           rows = int(ds.Rows)
           cols = int(ds.Columns)
           num_slices = len(sorted_files)
           pixel_spacing = [float(x) for x in getattr(ds, 'PixelSpacing', [1.0, 1.0])]
           # Compute slice spacing from positions if multiple slices
           if num_slices > 1:
               ds0 = pydicom.dcmread(str(sorted_files[0]), stop_before_pixels=True)
               ds1 = pydicom.dcmread(str(sorted_files[1]), stop_before_pixels=True)
               pos0 = np.array([float(x) for x in ds0.ImagePositionPatient])
               pos1 = np.array([float(x) for x in ds1.ImagePositionPatient])
               slice_spacing = float(np.linalg.norm(pos1 - pos0))
           else:
               slice_spacing = float(getattr(ds, 'SliceThickness', 1.0))
           voxel_spacing = [pixel_spacing[0], pixel_spacing[1], slice_spacing]
           modality = getattr(ds, 'Modality', 'unknown')
           study_desc = getattr(ds, 'StudyDescription', '')
           series_desc = getattr(ds, 'SeriesDescription', '')
           bits = int(getattr(ds, 'BitsAllocated', 16))
           signed = bool(getattr(ds, 'PixelRepresentation', 0))
           if bits == 16:
               dtype_str = "int16" if signed else "uint16"
           elif bits == 8:
               dtype_str = "int8" if signed else "uint8"
           else:
               dtype_str = f"int{bits}"
           return VolumeMetadata(
               id=str(uuid.uuid4()),
               filename=sorted_files[0].parent.name,
               path=str(sorted_files[0].parent),
               format="dicom",
               dimensions=[cols, rows, num_slices],
               voxel_spacing=voxel_spacing,
               dtype=dtype_str,
               modality=modality,
               study_description=study_desc,
               series_description=series_desc,
               num_files=num_slices,
           )
       ```

       import numpy as np

       def load_nifti_volume(filepath: Path) -> tuple[np.ndarray, dict]:
           img = nib.load(str(filepath))
           data = np.ascontiguousarray(img.get_fdata(dtype=np.float32))
           header = img.header
           dims = list(img.shape[:3])
           spacing = [float(s) for s in header.get_zooms()[:3]]
           dtype_str = str(header.get_data_dtype())
           min_val = float(np.min(data))
           modality = "CT" if min_val < -500 else "MR"
           stat = Path(filepath).stat()
           file_date = datetime.fromtimestamp(stat.st_mtime, tz=timezone.utc).strftime("%Y-%m-%d")
           metadata = {
               "dimensions": dims,
               "voxel_spacing": spacing,
               "dtype": dtype_str,
               "modality": modality,
               "file_date": file_date,
           }
           return data, metadata

    5. Create server/loaders/dicom_loader.py:
       ```python
       from pathlib import Path
       import numpy as np
       import pydicom
       from catalog.dicom_grouper import sort_dicom_slices

       def load_dicom_volume(dicom_files: list[Path]) -> tuple[np.ndarray, dict]:
           sorted_files = sort_dicom_slices(dicom_files) if len(dicom_files) > 1 else dicom_files
           slices = []
           for f in sorted_files:
               ds = pydicom.dcmread(str(f))
               arr = ds.pixel_array.astype(np.float32)
               slope = float(getattr(ds, 'RescaleSlope', 1.0))
               intercept = float(getattr(ds, 'RescaleIntercept', 0.0))
               arr = arr * slope + intercept
               slices.append(arr)
           volume = np.stack(slices, axis=-1)  # shape: (rows, cols, num_slices)
           volume = np.ascontiguousarray(volume)
           ds0 = pydicom.dcmread(str(sorted_files[0]), stop_before_pixels=True)
           pixel_spacing = [float(x) for x in getattr(ds0, 'PixelSpacing', [1.0, 1.0])]
           if len(sorted_files) > 1:
               ds1 = pydicom.dcmread(str(sorted_files[1]), stop_before_pixels=True)
               pos0 = np.array([float(x) for x in ds0.ImagePositionPatient])
               pos1 = np.array([float(x) for x in ds1.ImagePositionPatient])
               slice_spacing = float(np.linalg.norm(pos1 - pos0))
           else:
               slice_spacing = float(getattr(ds0, 'SliceThickness', 1.0))
           metadata = {
               "dimensions": [int(ds0.Columns), int(ds0.Rows), len(sorted_files)],
               "voxel_spacing": [pixel_spacing[0], pixel_spacing[1], slice_spacing],
               "dtype": "float32",
               "modality": getattr(ds0, 'Modality', 'unknown'),
               "study_description": getattr(ds0, 'StudyDescription', ''),
               "series_description": getattr(ds0, 'SeriesDescription', ''),
           }
           return volume, metadata
       ```
