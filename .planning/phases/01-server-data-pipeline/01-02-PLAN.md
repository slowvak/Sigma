    2. Create server/api/volumes.py with an APIRouter:
       ```python
       from fastapi import APIRouter, Request, Response, HTTPException
       from pathlib import Path
       import numpy as np
       from catalog.models import VolumeMetadata
       from loaders.nifti_loader import load_nifti_volume
       from loaders.dicom_loader import load_dicom_volume

       router = APIRouter(prefix="/api/volumes", tags=["volumes"])

       @router.get("")
       async def list_volumes(request: Request) -> list[VolumeMetadata]:
           return request.app.state.volumes

       @router.get("/{volume_id}/metadata")
       async def get_volume_metadata(volume_id: str, request: Request) -> VolumeMetadata:
           for vol in request.app.state.volumes:
               if vol.id == volume_id:
                   return vol
           raise HTTPException(status_code=404, detail="Volume not found")

       @router.get("/{volume_id}/data")
       async def get_volume_data(volume_id: str, request: Request):
           vol = None
           for v in request.app.state.volumes:
               if v.id == volume_id:
                   vol = v
                   break
           if vol is None:
               raise HTTPException(status_code=404, detail="Volume not found")
           # Load on demand
           if vol.format == "nifti":
               data, _ = load_nifti_volume(Path(vol.path))
           elif vol.format == "dicom":
               # Collect DICOM files from the volume path directory
               dicom_dir = Path(vol.path)
               dicom_files = sorted(dicom_dir.glob("*"))
               data, _ = load_dicom_volume(dicom_files)
           else:
               raise HTTPException(status_code=400, detail=f"Unknown format: {vol.format}")
           data = np.ascontiguousarray(data, dtype=np.float32)
           return Response(
               content=data.tobytes(),
               media_type="application/octet-stream",
               headers={
                   "Content-Length": str(len(data.tobytes())),
               }
           )
       ```

    3. Create server/main.py using the modern lifespan context manager (NOT deprecated @app.on_event):
       ```python
       import sys
       from pathlib import Path
       from contextlib import asynccontextmanager
       from fastapi import FastAPI
       from fastapi.middleware.cors import CORSMiddleware

       _data_folder: str | None = None

       @asynccontextmanager
       async def lifespan(app: FastAPI):
           from catalog.scanner import scan_folder
           folder = Path(_data_folder) if _data_folder else Path(".")
           app.state.volumes = await scan_folder(folder)
           # Store file lists for DICOM volumes (needed for on-demand loading)
           app.state.dicom_file_map = {}
           print(f"Cataloged {len(app.state.volumes)} volumes from {folder}")
           yield

       app = FastAPI(title="NextEd Server", lifespan=lifespan)
       app.add_middleware(
           CORSMiddleware,
           allow_origins=["*"],
           allow_methods=["*"],
           allow_headers=["*"],
       )

       from api.volumes import router as volumes_router
       app.include_router(volumes_router)

       def main():
           import uvicorn
           global _data_folder
           if len(sys.argv) < 2:
               print("Usage: python -m main <folder_path>")
               sys.exit(1)
           _data_folder = sys.argv[1]
           if not Path(_data_folder).is_dir():
               print(f"Error: {_data_folder} is not a directory")
               sys.exit(1)
           uvicorn.run("main:app", host="0.0.0.0", port=8000, reload=True)

       if __name__ == "__main__":
           main()
       ```
