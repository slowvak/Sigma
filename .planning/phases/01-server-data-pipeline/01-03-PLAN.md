       ```javascript
       const API_BASE = '/api';

       export async function fetchVolumes() {
           const response = await fetch(`${API_BASE}/volumes`);
           if (!response.ok) throw new Error(`Failed to fetch volumes: ${response.status}`);
           return response.json();
       }

       export async function fetchVolumeMetadata(volumeId) {
           const response = await fetch(`${API_BASE}/volumes/${volumeId}/metadata`);
           if (!response.ok) throw new Error(`Failed to fetch metadata: ${response.status}`);
           return response.json();
       }

       export async function fetchVolumeData(volumeId) {
           const response = await fetch(`${API_BASE}/volumes/${volumeId}/data`);
           if (!response.ok) throw new Error(`Failed to fetch volume data: ${response.status}`);
           return response.arrayBuffer();
       }
       ```

    6. Create client/src/ui/volumeList.js:
       ```javascript
       export function renderVolumeList(volumes, container, onSelect) {
           container.innerHTML = '';
           if (volumes.length === 0) {
               const li = document.createElement('li');
               li.className = 'volume-item';
               li.textContent = 'No volumes found.';
               container.appendChild(li);
               return;
           }
           for (const vol of volumes) {
               const li = document.createElement('li');
               li.className = 'volume-item';
               li.setAttribute('role', 'option');
               li.setAttribute('aria-selected', 'false');
               li.setAttribute('tabindex', '0');
               li.dataset.volumeId = vol.id;

               const headerRow = document.createElement('div');
               headerRow.className = 'volume-item-header';

               const name = document.createElement('span');
               name.className = 'volume-name';
               name.textContent = vol.filename;

               const badge = document.createElement('span');
               badge.className = `volume-badge ${vol.format}`;
               badge.textContent = vol.format.toUpperCase();

               headerRow.appendChild(name);
               headerRow.appendChild(badge);

               const dims = document.createElement('span');
               dims.className = 'volume-dims';
               dims.textContent = `${vol.dimensions[0]} x ${vol.dimensions[1]} x ${vol.dimensions[2]}`;

               li.appendChild(headerRow);
               li.appendChild(dims);

               li.addEventListener('click', () => {
                   // Deselect all
                   container.querySelectorAll('.volume-item').forEach(item => {
                       item.classList.remove('selected');
                       item.setAttribute('aria-selected', 'false');
                   });
                   li.classList.add('selected');
                   li.setAttribute('aria-selected', 'true');
                   onSelect(vol);
               });

               container.appendChild(li);
           }
       }
       ```

    7. Create client/src/ui/volumeDetail.js:
       ```javascript
       import { fetchVolumeData } from '../api.js';

       export function renderEmptyState(container) {
           container.innerHTML = '';
           const empty = document.createElement('div');
           empty.className = 'empty-state';
           const h2 = document.createElement('h2');
           h2.textContent = 'Select a Volume';
           const p = document.createElement('p');
           p.textContent = 'Choose a volume from the list to view its details.';
           empty.appendChild(h2);
           empty.appendChild(p);
           container.appendChild(empty);
       }

       export function renderVolumeDetail(volume, container) {
           container.innerHTML = '';

           const heading = document.createElement('h2');
           heading.className = 'detail-heading';
           heading.textContent = 'Volume Details';
           container.appendChild(heading);

           // Common fields
           const fields = [
               ['Filename', volume.filename],
               ['Format', volume.format.toUpperCase()],
               ['Dimensions', `${volume.dimensions[0]} x ${volume.dimensions[1]} x ${volume.dimensions[2]}`],
               ['Voxel Spacing', `${volume.voxel_spacing[0]} x ${volume.voxel_spacing[1]} x ${volume.voxel_spacing[2]} mm`],
               ['Modality', volume.modality],
           ];

           // DICOM-specific fields (BROW-02)
           if (volume.format === 'dicom') {
               if (volume.study_description) {
                   fields.push(['Study Description', volume.study_description]);
               }
               if (volume.series_description) {
                   fields.push(['Series Description', volume.series_description]);
               }
               fields.push(['Number of Files', String(volume.num_files)]);
           }

           // NIfTI-specific fields (BROW-03)
           if (volume.format === 'nifti' && volume.file_date) {
               fields.push(['File Date', volume.file_date]);
           }

           for (const [label, value] of fields) {
               const fieldDiv = document.createElement('div');
               fieldDiv.className = 'detail-field';

               const labelEl = document.createElement('div');
               labelEl.className = 'detail-label';
               labelEl.textContent = label;

               const valueEl = document.createElement('div');
               valueEl.className = 'detail-value';
               valueEl.textContent = value;

               fieldDiv.appendChild(labelEl);
               fieldDiv.appendChild(valueEl);
               container.appendChild(fieldDiv);
           }

           // Open Volume button (BROW-04)
           const btn = document.createElement('button');
           btn.className = 'btn-open';
           btn.textContent = 'Open Volume';
           btn.addEventListener('click', async () => {
               btn.disabled = true;
               btn.textContent = 'Loading...';
               try {
                   const buffer = await fetchVolumeData(volume.id);
                   console.log(`Volume loaded: ${volume.filename}, ArrayBuffer size: ${buffer.byteLength}`);
                   btn.textContent = 'Loaded';
                   btn.classList.add('loaded');
                   setTimeout(() => {
                       btn.textContent = 'Open Volume';
                       btn.classList.remove('loaded');
                       btn.disabled = false;
                   }, 2000);
               } catch (err) {
                   console.error('Failed to load volume:', err);
                   btn.textContent = 'Failed to load';
                   btn.disabled = false;
                   setTimeout(() => {
                       btn.textContent = 'Open Volume';
                   }, 3000);
               }
           });
           container.appendChild(btn);
       }
       ```

    8. Create client/src/main.js:
       ```javascript
       import './styles.css';
       import { createAppShell } from './ui/appShell.js';
       import { renderVolumeList } from './ui/volumeList.js';
       import { renderVolumeDetail, renderEmptyState } from './ui/volumeDetail.js';
       import { fetchVolumes } from './api.js';

       async function init() {
           const { listContainer, detailPanel } = createAppShell();
           renderEmptyState(detailPanel);

           try {
               const volumes = await fetchVolumes();
               renderVolumeList(volumes, listContainer, (volume) => {
                   renderVolumeDetail(volume, detailPanel);
               });
               if (volumes.length === 0) {
                   detailPanel.innerHTML = '';
                   const banner = document.createElement('div');
                   banner.className = 'error-banner';
                   banner.textContent = 'No volumes found. Check that the server data folder contains NIfTI (.nii, .nii.gz) or DICOM (.dcm) files.';
                   detailPanel.appendChild(banner);
               }
           } catch (err) {
               console.error('Failed to fetch volumes:', err);
               detailPanel.innerHTML = '';
               const banner = document.createElement('div');
               banner.className = 'error-banner';
               banner.textContent = 'Cannot connect to server. Make sure the NextEd server is running on port 8000.';
               detailPanel.appendChild(banner);
           }
       }

       init();
       ```
